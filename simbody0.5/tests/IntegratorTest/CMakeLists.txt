FILE(GLOB INCLUDE_FILES *.h)

   TARGET_LINK_LIBRARIES(IntegratorTest       ${SHARED_TARGET} pthread )
   TARGET_LINK_LIBRARIES(IntegratorTestStatic ${STATIC_TARGET} pthread )
ENDIF( WIN32 )
